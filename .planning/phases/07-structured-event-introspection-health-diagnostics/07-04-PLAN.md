**Change 3: src/tests/health-shape.test.ts — frozen fixture comparison**

Create a new test file that loads `anomaly-record-v1.json` and verifies no field is added or removed from the `HealthAnomaly` shape without an explicit fixture update.

```typescript
import { readFileSync } from "node:fs";
import { dirname, join } from "node:path";
import { fileURLToPath } from "node:url";
import { describe, expect, it } from "vitest";
import type { HealthAnomaly } from "../index.js";

const rootDir = dirname(dirname(dirname(fileURLToPath(import.meta.url))));

function loadFixture(): unknown[] {
  return JSON.parse(
    readFileSync(join(rootDir, "src/tests/fixtures/anomaly-record-v1.json"), "utf8")
  ) as unknown[];
}

describe("anomaly-record-v1.json frozen fixture", () => {
  it("has exactly four records (one per AnomalyCode)", () => {
    const records = loadFixture();
    expect(records).toHaveLength(4);
  });

  it("each record has all required HealthAnomaly fields", () => {
    const records = loadFixture() as HealthAnomaly[];
    const requiredFields = ["code", "severity", "value", "threshold"] as const;
    for (const record of records) {
      for (const field of requiredFields) {
        expect(record).toHaveProperty(field);
      }
    }
  });

  it("budget-near-miss record has no agentId (global anomaly)", () => {
    const records = loadFixture() as HealthAnomaly[];
    const nearMiss = records.find(r => r.code === "budget-near-miss");
    expect(nearMiss).toBeDefined();
    expect(nearMiss).not.toHaveProperty("agentId");
  });

  it("per-agent records have agentId", () => {
    const records = loadFixture() as HealthAnomaly[];
    const perAgentCodes = ["empty-contribution", "provider-error-recovered", "runaway-turns"] as const;
    for (const code of perAgentCodes) {
      const record = records.find(r => r.code === code);
      expect(record, `record for ${code}`).toBeDefined();
      expect(record, `agentId on ${code}`).toHaveProperty("agentId");
    }
  });

  it("severity values match expected pattern", () => {
    const records = loadFixture() as HealthAnomaly[];
    const severityMap: Record<string, "warning" | "error"> = {
      "budget-near-miss": "warning",
      "empty-contribution": "error",
      "provider-error-recovered": "warning",
      "runaway-turns": "error"
    };
    for (const record of records) {
      expect(record.severity).toBe(severityMap[record.code]);
    }
  });
});
```
