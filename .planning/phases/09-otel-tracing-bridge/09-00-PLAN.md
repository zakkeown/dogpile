**1) Append to `src/testing/deterministic-provider.ts`** the following new export (do NOT modify the existing `createDeterministicModelProvider` — add at end of file, before the existing private helpers `readStringMetadata`/`tokenEstimate`):

```typescript
export interface DelegatingProviderOptions {
  readonly id?: string;
  readonly childProtocol?: "sequential" | "broadcast" | "coordinator";
  readonly childIntent?: string;
}

/**
 * Deterministic provider that drives a live coordinator sub-run dispatch end
 * to end. Used by Phase 9 OTEL contract tests to verify the
 * `RunProtocolOptions.parentSpan` + `subRunSpansByChildId` threading without
 * synthesizing events.
 *
 * On the FIRST `plan`-phase invocation under the coordinator protocol, the
 * provider emits a `delegate:` block (parsed by the runtime as a delegate
 * decision, triggering a child runProtocol call). On the SECOND `plan`-phase
 * invocation it emits a participate block, terminating the dispatch loop.
 *
 * `worker` and `final-synthesis` phases return safe deterministic text.
 *
 * Child sub-run model calls are answered by the existing
 * `createDeterministicModelProvider` text-shape; this provider serves both
 * parent and child runProtocol invocations from a single instance.
 */
export function createDelegatingDeterministicProvider(
  opts: DelegatingProviderOptions = {}
): ConfiguredModelProvider {
  const id = opts.id ?? "deterministic-delegating-model";
  const childProtocol = opts.childProtocol ?? "sequential";
  const childIntent = opts.childIntent ?? "delegated child run";

  const delegateBlock = JSON.stringify({ protocol: childProtocol, intent: childIntent });
  const delegateText = ["delegate:", "```json", delegateBlock, "```", ""].join("\n");
  const participateText = [
    "role_selected: coordinator",
    "participation: contribute",
    "rationale: synthesize after sub-run",
    "contribution:",
    "synthesized after sub-run"
  ].join("\n");

  let parentPlanIndex = 0;

  return {
    id,
    async generate(request: ModelRequest): Promise<ModelResponse> {
      const phase = readStringMetadata(request, "phase");
      const protocol = readStringMetadata(request, "protocol");
      const role = readStringMetadata(request, "role");
      const agentId = readStringMetadata(request, "agentId");

      // Top-level coordinator parent: alternate delegate then participate.
      if (protocol === "coordinator" && phase === "plan") {
        const text = parentPlanIndex === 0 ? delegateText : participateText;
        parentPlanIndex += 1;
        return {
          text,
          usage: { inputTokens: 1, outputTokens: 1, totalTokens: 2 },
          costUsd: 0
        };
      }

      if (protocol === "coordinator" && phase === "worker") {
        return {
          text: "worker output",
          usage: { inputTokens: 1, outputTokens: 1, totalTokens: 2 },
          costUsd: 0
        };
      }

      if (protocol === "coordinator" && phase === "final-synthesis") {
        return {
          text: `${role}:${agentId} synthesized the coordinator-managed mission.`,
          usage: { inputTokens: 1, outputTokens: 1, totalTokens: 2 },
          costUsd: 0
        };
      }

      // Child run (sequential / broadcast / nested coordinator) — return safe
      // deterministic text in the established shape so the child run completes
      // without further delegation.
      const text = `${role}:${agentId} framed the mission.`;
      return {
        text,
        usage: { inputTokens: 1, outputTokens: 1, totalTokens: 2 },
        costUsd: 0
      };
    }
  };
}
```

**2) Create `src/testing/deterministic-provider.test.ts`:**

```typescript
import { describe, expect, it } from "vitest";
import { run } from "../runtime/engine.js";
import { createDelegatingDeterministicProvider } from "./deterministic-provider.js";
import type { SubRunStartedEvent, SubRunCompletedEvent } from "../types/events.js";

describe("createDelegatingDeterministicProvider", () => {
  it("drives a live coordinator sub-run dispatch (sub-run-started + sub-run-completed events emitted)", async () => {
    const result = await run({
      intent: "verify delegating provider drives sub-run dispatch",
      protocol: { kind: "coordinator", maxTurns: 4 },
      tier: "fast",
      model: createDelegatingDeterministicProvider(),
      agents: [
        { id: "lead", role: "coordinator" },
        { id: "worker-a", role: "worker" }
      ]
    });

    const started = result.trace.events.filter(
      (e): e is SubRunStartedEvent => e.type === "sub-run-started"
    );
    const completed = result.trace.events.filter(
      (e): e is SubRunCompletedEvent => e.type === "sub-run-completed"
    );

    expect(started.length).toBeGreaterThanOrEqual(1);
    expect(completed.length).toBeGreaterThanOrEqual(1);
    // The first sub-run-started must be paired with a sub-run-completed of the same childRunId.
    const startedIds = new Set(started.map((e) => e.childRunId));
    const completedIds = new Set(completed.map((e) => e.childRunId));
    for (const childId of startedIds) {
      expect(completedIds.has(childId)).toBe(true);
    }
  });

  it("respects the childProtocol option", async () => {
    const result = await run({
      intent: "verify childProtocol option propagates to dispatched sub-run",
      protocol: { kind: "coordinator", maxTurns: 4 },
      tier: "fast",
      model: createDelegatingDeterministicProvider({ childProtocol: "sequential" }),
      agents: [
        { id: "lead", role: "coordinator" },
        { id: "worker-a", role: "worker" }
      ]
    });

    const started = result.trace.events.find(
      (e) => e.type === "sub-run-started"
    ) as SubRunStartedEvent | undefined;
    expect(started).toBeDefined();
    // Implementations of SubRunStartedEvent carry the child protocol; verify by
    // searching the trace for a child run-started downstream of the dispatch.
  });

  it("emits a delegate text block on the first plan phase and a participate block on the second", async () => {
    const provider = createDelegatingDeterministicProvider({ id: "phase-test" });
    const first = await provider.generate({
      messages: [{ role: "user", content: "anything" }],
      metadata: { phase: "plan", protocol: "coordinator", role: "coordinator", agentId: "lead" }
    } as Parameters<typeof provider.generate>[0]);
    const second = await provider.generate({
      messages: [{ role: "user", content: "anything" }],
      metadata: { phase: "plan", protocol: "coordinator", role: "coordinator", agentId: "lead" }
    } as Parameters<typeof provider.generate>[0]);

    expect(first.text).toContain("delegate:");
    expect(first.text).toContain("\"protocol\":\"sequential\"");
    expect(second.text).not.toContain("delegate:");
    expect(second.text).toContain("participation: contribute");
  });
});
```
