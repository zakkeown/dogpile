**1) Determine span name choice for top-level vs sub-run.**

A top-level run uses `dogpile.run`. A sub-run started by a coordinator child dispatch — the child's own `runNonStreamingProtocol` invocation — also opens a `dogpile.run` span, but its parent is the parent's `dogpile.sub-run` span (passed via `options.parentSpan`). The `dogpile.sub-run` span itself is opened in the PARENT's emit closure when `sub-run-started` is observed; the child's `dogpile.run` span is the inner span.

So at runNonStreamingProtocol entry, always open `dogpile.run` (regardless of depth). Use `options.parentSpan` as the parent — it is the parent's sub-run span when this is a child invocation, undefined when top-level.

**2) Critical agent-turn span lifecycle (per D-08):**

D-08 requires the `dogpile.agent-turn` span to cover `ModelRequestEvent.startedAt → TurnEvent.at`. This means:
- **OPEN the agent-turn span on the FIRST `model-request` event for a given agentId** (when `agentTurnSpans` does not yet have an entry for that agentId). Store in `agentTurnSpans.set(event.agentId, span)`.
- **CLOSE the agent-turn span on the `agent-turn` event** for that agentId. Read from `agentTurnSpans.get(event.agentId)`, set end-time attributes (cost, tokens), `setStatus("ok")`, `end()`, then `delete`.
- DO NOT call `tracer.startSpan(DOGPILE_SPAN_NAMES.AGENT_TURN, ...)` in the `case "agent-turn"` block — that produced zero-duration spans and broke the D-07 hierarchy because `agentTurnSpans` was never populated.
- The `model-call` span MUST be parented under the open agent-turn span (read `agentTurnSpans.get(event.agentId)` first, then fall back to sub-run/run span). This enforces D-07: `run → sub-run → agent-turn → model-call`.

```typescript
async function runNonStreamingProtocol(options: NonStreamingProtocolOptions): Promise<RunResult> {
  const failureInstancesByChildRunId = new Map<string, DogpileError>();
  const abortLifecycle = createNonStreamingAbortLifecycle({
    callerSignal: options.signal,
    timeoutMs: runtimeTimeoutMs(options),
    providerId: options.model.id,
    timeoutErrorSource:
      options.currentDepth !== undefined &&
      options.currentDepth > 0 &&
      options.parentDeadlineMs === undefined
        ? "engine"
        : "runtime"
  });

  // --- TRACING (Phase 9 / OTEL-01..OTEL-03) ---
  const tracer = options.tracer;
  const runSpan: DogpileSpan | undefined = tracer?.startSpan(DOGPILE_SPAN_NAMES.RUN, {
    ...(options.parentSpan ? { parent: options.parentSpan } : {}),
    attributes: {
      "dogpile.run.protocol": options.protocol.kind,
      "dogpile.run.tier": String(options.tier),
      "dogpile.run.intent": options.intent.slice(0, 200),
    }
  });

  // Span state maps (only allocated when tracer is present)
  const subRunSpans = tracer ? new Map<string, DogpileSpan>() : undefined;
  const agentTurnSpans = tracer ? new Map<string, DogpileSpan>() : undefined;
  const modelCallSpans = tracer ? new Map<string, DogpileSpan>() : undefined;
  const pendingModelRequests = tracer ? new Map<string, ModelRequestEvent>() : undefined;
  const agentTurnCounters = tracer ? new Map<string, number>() : undefined;
  const tokenAccumByAgent = tracer ? new Map<string, { inputTokens: number; outputTokens: number }>() : undefined;
  // ----------------------------------------------

  let runOutcome: "completed" | "budget-stopped" | "aborted" = "completed";
  let runTerminationReason: string | undefined;
  let runAgentCount = 0;
  let runTurnCount = 0;
  let runCostUsd = 0;
  let runInputTokens = 0;
  let runOutputTokens = 0;

  try {
    const emittedEvents: RunEvent[] = [];
    const result = await abortLifecycle.run(runProtocol({
      ...options,
      ...(abortLifecycle.signal !== undefined ? { signal: abortLifecycle.signal } : {}),
      ...(runSpan ? { parentSpan: runSpan } : {}),
      ...(subRunSpans ? { subRunSpansByChildId: subRunSpans } : {}),
      emit(event: RunEvent): void {
        emittedEvents.push(event);

        // Zero-overhead fast path: any work below this guard runs only when tracer is present.
        if (!tracer || !subRunSpans || !agentTurnSpans || !modelCallSpans || !pendingModelRequests || !agentTurnCounters || !tokenAccumByAgent) return;

        switch (event.type) {
          case "model-request": {
            pendingModelRequests.set(event.agentId, event);

            // Open the agent-turn span on the FIRST model-request for this agentId
            // (covers D-08: span spans ModelRequestEvent.startedAt → TurnEvent.at).
            if (!agentTurnSpans.has(event.agentId)) {
              const turnNumber = (agentTurnCounters.get(event.agentId) ?? 0) + 1;
              agentTurnCounters.set(event.agentId, turnNumber);
              const turnParent = subRunSpans.get(event.runId) ?? runSpan;
              const turnSpan = tracer.startSpan(DOGPILE_SPAN_NAMES.AGENT_TURN, {
                ...(turnParent ? { parent: turnParent } : {}),
                attributes: {
                  "dogpile.agent.id": event.agentId,
                  "dogpile.turn.number": turnNumber,
                  "dogpile.model.id": event.modelId,
                }
              });
              agentTurnSpans.set(event.agentId, turnSpan);
            }

            // Open the model-call span as a child of the open agent-turn span
            // (D-07: run → sub-run → agent-turn → model-call).
            const callParent =
              agentTurnSpans.get(event.agentId) ??
              subRunSpans.get(event.runId) ??
              runSpan;
            const callSpan = tracer.startSpan(DOGPILE_SPAN_NAMES.MODEL_CALL, {
              ...(callParent ? { parent: callParent } : {}),
              attributes: {
                "dogpile.model.id": event.modelId,
                "dogpile.call.id": event.callId,
                "dogpile.provider.id": event.providerId,
              }
            });
            modelCallSpans.set(event.callId, callSpan);
            break;
          }
          case "model-response": {
            const span = modelCallSpans.get(event.callId);
            if (span) {
              const inputTokens = event.response.usage?.inputTokens ?? 0;
              const outputTokens = event.response.usage?.outputTokens ?? 0;
              span.setAttribute("dogpile.model.input_tokens", inputTokens);
              span.setAttribute("dogpile.model.output_tokens", outputTokens);
              span.setStatus("ok");
              span.end();
              modelCallSpans.delete(event.callId);
              const accum = tokenAccumByAgent.get(event.agentId) ?? { inputTokens: 0, outputTokens: 0 };
              accum.inputTokens += inputTokens;
              accum.outputTokens += outputTokens;
              tokenAccumByAgent.set(event.agentId, accum);
            }
            break;
          }
          case "agent-turn": {
            // CLOSE the agent-turn span opened on the first model-request for this agentId.
            // Do NOT open a new span here — that produced zero-duration spans and broke D-07.
            const turnSpan = agentTurnSpans.get(event.agentId);
            if (turnSpan) {
              turnSpan.setAttribute("dogpile.agent.role", event.role);
              const accum = tokenAccumByAgent.get(event.agentId);
              if (accum) {
                turnSpan.setAttribute("dogpile.turn.input_tokens", accum.inputTokens);
                turnSpan.setAttribute("dogpile.turn.output_tokens", accum.outputTokens);
              }
              turnSpan.setAttribute("dogpile.turn.cost_usd", event.cost.usd);
              turnSpan.setStatus("ok");
              turnSpan.end();
              agentTurnSpans.delete(event.agentId);
            }
            pendingModelRequests.delete(event.agentId);
            tokenAccumByAgent.delete(event.agentId);
            break;
          }
          case "sub-run-started": {
            const parent = runSpan;
            const span = tracer.startSpan(DOGPILE_SPAN_NAMES.SUB_RUN, {
              ...(parent ? { parent } : {}),
              attributes: {
                "dogpile.sub_run.child_run_id": event.childRunId,
                "dogpile.sub_run.parent_run_id": event.parentRunId,
                "dogpile.sub_run.depth": event.depth,
              }
            });
            subRunSpans.set(event.childRunId, span);
            break;
          }
          case "sub-run-completed": {
            const span = subRunSpans.get(event.childRunId);
            if (span) {
              span.setStatus("ok");
              span.end();
              subRunSpans.delete(event.childRunId);
            }
            break;
          }
          case "sub-run-failed": {
            const span = subRunSpans.get(event.childRunId);
            if (span) {
              span.setStatus("error", event.error?.message ?? "sub-run failed");
              span.end();
              subRunSpans.delete(event.childRunId);
            }
            break;
          }
          default:
            // No span action for other event types in Phase 9.
            break;
        }
      },
      failureInstancesByChildRunId
    }));
    const events = emittedEvents.length > 0 ? emittedEvents : result.trace.events;
    const trace = {
      ...result.trace,
      events,
      budgetStateChanges: createReplayTraceBudgetStateChanges(events),
      finalOutput: createReplayTraceFinalOutput(result.output, events.at(-1) ?? result.trace.events.at(-1)!)
    };

    const runResult = {
      ...result,
      accounting: createRunAccounting({
        tier: trace.tier,
        ...(trace.budget.caps ? { budget: trace.budget.caps } : {}),
        ...(trace.budget.termination ? { termination: trace.budget.termination } : {}),
        cost: result.cost,
        events
      }),
      eventLog: createRunEventLog(trace.runId, trace.protocol, events),
      trace,
      health: computeHealth(trace, DEFAULT_HEALTH_THRESHOLDS)
    };
    const terminalThrow = resolveRuntimeTerminalThrow(runResult.trace, failureInstancesByChildRunId);
    if (terminalThrow) {
      // Treat terminal-throw as error for span status; rethrow path handled in catch below.
      throw terminalThrow;
    }

    const finalResult = canonicalizeRunResult(await abortLifecycle.run(applyRunEvaluation(runResult, options.evaluate)));

    // Derive run-level end-time attributes from the final canonical result.
    if (runSpan) {
      runAgentCount = finalResult.trace.agents?.length ?? new Set(finalResult.trace.events.filter(e => e.type === "agent-turn").map(e => (e as { agentId: string }).agentId)).size;
      runTurnCount = finalResult.trace.events.filter(e => e.type === "agent-turn").length;
      runCostUsd = finalResult.cost.usd;
      runInputTokens = finalResult.cost.inputTokens;
      runOutputTokens = finalResult.cost.outputTokens;
      const termination = finalResult.trace.budget?.termination;
      if (termination?.reason) {
        runOutcome = "budget-stopped";
        runTerminationReason = termination.reason;
      }
      runSpan.setAttribute("dogpile.run.id", finalResult.trace.runId);
      runSpan.setAttribute("dogpile.run.agent_count", runAgentCount);
      runSpan.setAttribute("dogpile.run.turn_count", runTurnCount);
      runSpan.setAttribute("dogpile.run.cost_usd", runCostUsd);
      runSpan.setAttribute("dogpile.run.input_tokens", runInputTokens);
      runSpan.setAttribute("dogpile.run.output_tokens", runOutputTokens);
      runSpan.setAttribute("dogpile.run.outcome", runOutcome);
      if (runTerminationReason !== undefined) {
        runSpan.setAttribute("dogpile.run.termination_reason", runTerminationReason);
      }
      runSpan.setStatus("ok");
      runSpan.end();
    }

    return finalResult;
  } catch (error: unknown) {
    if (runSpan) {
      const isAbort = error instanceof Error && (error.name === "AbortError" || error.message.includes("aborted"));
      runOutcome = isAbort ? "aborted" : "completed";
      runSpan.setAttribute("dogpile.run.outcome", isAbort ? "aborted" : "completed");
      runSpan.setStatus("error", error instanceof Error ? error.message : String(error));
      // Best-effort cleanup of any still-open child spans
      for (const span of (subRunSpans?.values() ?? [])) span.end();
      for (const span of (modelCallSpans?.values() ?? [])) span.end();
      for (const span of (agentTurnSpans?.values() ?? [])) span.end();
      runSpan.end();
    }
    throw abortLifecycle.translateError(error);
  } finally {
    failureInstancesByChildRunId.clear();
    abortLifecycle.cleanup();
  }
}
```
