async function runNonStreamingProtocol(options: NonStreamingProtocolOptions): Promise<RunResult> {
  const failureInstancesByChildRunId = new Map<string, DogpileError>();
  const abortLifecycle = createNonStreamingAbortLifecycle({ ... });
  const startedAtMs = Date.now();
  const effectiveLogger = options.logger ?? console;
  const subRunStartMs = new Map<string, number>();
  let finalResult: RunResult | undefined;

  function fireHook(cb: ((s: RunMetricsSnapshot) => void | Promise<void>) | undefined, snapshot: RunMetricsSnapshot): void {
    if (!cb) return;
    const ret = cb(snapshot);
    if (ret && typeof ret === "object" && "catch" in ret) {
      (ret as Promise<void>).catch((err: unknown) => {
        effectiveLogger.error("dogpile:metricsHook threw", {
          error: err instanceof Error ? err.message : String(err)
        });
      });
    }
  }

  try {
    const emittedEvents: RunEvent[] = [];
    const result = await abortLifecycle.run(runProtocol({
      ...options,
      ...(abortLifecycle.signal !== undefined ? { signal: abortLifecycle.signal } : {}),
      emit(event: RunEvent): void {
        emittedEvents.push(event);
        // Sub-run start tracking for durationMs (D-06)
        if (event.type === "sub-run-started" && options.metricsHook?.onSubRunComplete) {
          subRunStartMs.set(event.childRunId, Date.now());
        }
        // Sub-run completion: fire onSubRunComplete
        if (event.type === "sub-run-completed" && options.metricsHook?.onSubRunComplete) {
          const subStartMs = subRunStartMs.get(event.childRunId) ?? Date.now();
          subRunStartMs.delete(event.childRunId);
          const snapshot = buildSubRunSnapshot(event.subResult, Date.now() - subStartMs);
          fireHook(options.metricsHook.onSubRunComplete, snapshot);
        }
      },
      failureInstancesByChildRunId
    }));
    // ... existing result assembly code (DO NOT CHANGE) ...
    const runResult = canonicalizeRunResult(await abortLifecycle.run(applyRunEvaluation(runResult_assembled, options.evaluate)));
    finalResult = runResult;
    // Fire onRunComplete for the happy path (completed or budget-stopped)
    if (options.metricsHook?.onRunComplete) {
      fireHook(options.metricsHook.onRunComplete, buildRunSnapshot(finalResult, Date.now() - startedAtMs));
    }
    return finalResult;
  } catch (error: unknown) {
    throw abortLifecycle.translateError(error);
  } finally {
    // For aborted runs: finalResult is undefined (catch threw before assignment)
    // D-10: hook fires for ALL terminal states; for aborted we fire in finally
    // but only if onRunComplete hasn't fired yet (finalResult === undefined means we're in error path)
    if (options.metricsHook?.onRunComplete && finalResult === undefined) {
      // Build a partial snapshot from what we have — minimal aborted snapshot
      const abortedSnapshot: RunMetricsSnapshot = {
        outcome: "aborted",
        inputTokens: 0,
        outputTokens: 0,
        costUsd: 0,
        totalInputTokens: 0,
        totalOutputTokens: 0,
        totalCostUsd: 0,
        turns: 0,
        durationMs: Date.now() - startedAtMs
      };
      fireHook(options.metricsHook.onRunComplete, abortedSnapshot);
    }
    subRunStartMs.clear();
    failureInstancesByChildRunId.clear();
    abortLifecycle.cleanup();

}
